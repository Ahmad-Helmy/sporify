   random (array) {
    for (var i = array.length - 1; i > 0; i--) {
        var j = Math.floor(Math.random() * (i + 1));
        var temp = array[i];
        array[i] = array[j];

    }
}
  /**
   * to mute the audio
   */
  mute() {
    if (this.muted) {
      this.soundon = false;
      this.x.volume = this.vol;
      this.muted = false;
    } else {
      this.soundon = true;
      this.x.volume = 0;
      this.muted = true;
    }
  }
  /**
   * toggle like button
   */
  like() {


    if (this.isliked == true) {
      this.isliked = false;
    }
    else this.isliked = true;

  }
